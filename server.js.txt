const express = require('express');
const cors = require('cors');
const path = require('path');
const fs = require('fs');

const app = express();
app.use(express.json());
app.use(cors());
app.use(express.static('public'));

// In-memory storage
let orders = [];
let orderCounter = 0;

// Load data from file
const loadData = () => {
  try {
    const data = fs.readFileSync('orders.json', 'utf8');
    orders = JSON.parse(data);
    orderCounter = orders.length;
  } catch (e) {
    orders = [];
  }
};

// Save data to file
const saveData = () => {
  fs.writeFileSync('orders.json', JSON.stringify(orders, null, 2));
};

// Load on startup
loadData();

// 1. CREATE ORDER
app.post('/api/orders', (req, res) => {
  const { customerName, phone, items } = req.body;
  
  // Validation
  if (!customerName || !phone || !items || !Array.isArray(items)) {
    return res.status(400).json({ error: 'Missing required fields' });
  }
  
  if (!/^\d{10}$/.test(phone)) {
    return res.status(400).json({ error: 'Invalid phone number' });
  }
  
  // Calculate total
  const total = items.reduce((sum, item) => {
    return sum + (item.quantity * item.price);
  }, 0);
  
  // Create order
  const orderId = `ORD${String(++orderCounter).padStart(3, '0')}`;
  const order = {
    id: orderId,
    customerName,
    phone,
    items,
    total,
    status: 'RECEIVED',
    createdAt: new Date().toISOString()
  };
  
  orders.push(order);
  saveData();
  
  res.json({ 
    success: true, 
    orderId, 
    total,
    status: 'RECEIVED' 
  });
});

// 2. UPDATE STATUS
app.patch('/api/orders/:id/status', (req, res) => {
  const { id } = req.params;
  const { status } = req.body;
  
  const validStatuses = ['RECEIVED', 'PROCESSING', 'READY', 'DELIVERED'];
  if (!validStatuses.includes(status)) {
    return res.status(400).json({ error: 'Invalid status' });
  }
  
  const orderIndex = orders.findIndex(o => o.id === id);
  if (orderIndex === -1) {
    return res.status(404).json({ error: 'Order not found' });
  }
  
  orders[orderIndex].status = status;
  saveData();
  
  res.json({ success: true, order: orders[orderIndex] });
});

// 3. GET ALL ORDERS (with filters)
app.get('/api/orders', (req, res) => {
  let filteredOrders = [...orders];
  
  const { status, phone, customerName } = req.query;
  
  if (status) {
    filteredOrders = filteredOrders.filter(o => o.status === status);
  }
  
  if (phone) {
    filteredOrders = filteredOrders.filter(o => o.phone.includes(phone));
  }
  
  if (customerName) {
    filteredOrders = filteredOrders.filter(o => 
      o.customerName.toLowerCase().includes(customerName.toLowerCase())
    );
  }
  
  res.json(filteredOrders);
});

// 4. DASHBOARD
app.get('/api/dashboard', (req, res) => {
  const totalOrders = orders.length;
  const totalRevenue = orders.reduce((sum, o) => sum + o.total, 0);
  
  const ordersByStatus = {
    RECEIVED: orders.filter(o => o.status === 'RECEIVED').length,
    PROCESSING: orders.filter(o => o.status === 'PROCESSING').length,
    READY: orders.filter(o => o.status === 'READY').length,
    DELIVERED: orders.filter(o => o.status === 'DELIVERED').length
  };
  
  res.json({
    totalOrders,
    totalRevenue,
    ordersByStatus
  });
});

app.listen(3000, () => {
  console.log('🚀 Laundry System running on http://localhost:3000');
  console.log('📊 Dashboard: http://localhost:3000/dashboard');
});
