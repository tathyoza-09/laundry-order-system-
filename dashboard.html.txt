<head>
    <title>Laundry Dashboard</title>
    <style>
        body { font-family: Arial; max-width: 800px; margin: 50px auto; padding: 20px; }
        .card { background: #f5f5f5; padding: 20px; margin: 20px 0; border-radius: 8px; }
        button { background: #007bff; color: white; padding: 10px 20px; border: none; border-radius: 5px; cursor: pointer; }
        input { padding: 10px; margin: 5px; border: 1px solid #ddd; border-radius: 4px; }
        table { width: 100%; border-collapse: collapse; margin-top: 20px; }
        th, td { padding: 12px; text-align: left; border-bottom: 1px solid #ddd; }
        .status { padding: 4px 8px; border-radius: 4px; color: white; font-size: 12px; }
    </style>
</head>
<body>
    <h1>🧺 Laundry Order Management</h1>
    
    <!-- CREATE ORDER FORM -->
    <div class="card">
        <h3>Create Order</h3>
        <input id="customerName" placeholder="Customer Name" />
        <input id="phone" placeholder="Phone (10 digits)" maxlength="10" />
        <input id="itemType" placeholder="Item (Shirt/Pants)" />
        <input id="quantity" type="number" placeholder="Quantity" />
        <input id="price" type="number" placeholder="Price per item" />
        <button onclick="createOrder()">Add Order</button>
    </div>
    
    <!-- DASHBOARD -->
    <div class="card">
        <h3>📊 Dashboard</h3>
        <div id="dashboard"></div>
        <button onclick="loadDashboard()">Refresh</button>
    </div>
    
    <!-- ORDERS TABLE -->
    <div class="card">
        <h3>📋 All Orders</h3>
        <input id="filter" placeholder="Filter by name/phone/status" oninput="loadOrders()" />
        <table id="ordersTable">
            <thead>
                <tr>
                    <th>ID</th>
                    <th>Customer</th>
                    <th>Total</th>
                    <th>Status</th>
                    <th>Actions</th>
                </tr>
            </thead>
            <tbody></tbody>
        </table>
    </div>

    <script>
        // Load dashboard
        async function loadDashboard() {
            const res = await fetch('/api/dashboard');
            const data = await res.json();
            document.getElementById('dashboard').innerHTML = `
                <p><strong>Total Orders:</strong> ${data.totalOrders}</p>
                <p><strong>Total Revenue:</strong> ₹${data.totalRevenue}</p>
                <p><strong>Status:</strong> Received:${data.ordersByStatus.RECEIVED} | 
                    Processing:${data.ordersByStatus.PROCESSING} | 
                    Ready:${data.ordersByStatus.READY} | 
                    Delivered:${data.ordersByStatus.DELIVERED}</p>
            `;
        }
        
        // Load orders
        async function loadOrders() {
            const filter = document.getElementById('filter').value;
            const url = filter ? `/api/orders?${new URLSearchParams({customerName: filter})}` : '/api/orders';
            const res = await fetch(url);
            const orders = await res.json();
            
            const tbody = document.querySelector('#ordersTable tbody');
            tbody.innerHTML = orders.map(order => `
                <tr>
                    <td>${order.id}</td>
                    <td>${order.customerName}<br><small>${order.phone}</small></td>
                    <td>₹${order.total}</td>
                    <td><span class="status status-${order.status.toLowerCase()}">${order.status}</span></td>
                    <td>
                        <select onchange="updateStatus('${order.id}', this.value)">
                            <option ${order.status === 'RECEIVED' ? 'selected' : ''}>RECEIVED</option>
                            <option ${order.status === 'PROCESSING' ? 'selected' : ''}>PROCESSING</option>
                            <option ${order.status === 'READY' ? 'selected' : ''}>READY</option>
                            <option ${order.status === 'DELIVERED' ? 'selected' : ''}>DELIVERED</option>
                        </select>
                    </td>
                </tr>
            `).join('');
        }
        
        // Create order
        async function createOrder() {
            const customerName = document.getElementById('customerName').value;
            const phone = document.getElementById('phone').value;
            const itemType = document.getElementById('itemType').value;
            const quantity = parseInt(document.getElementById('quantity').value);
            const price = parseFloat(document.getElementById('price').value);
            
            if (!customerName || !phone || !itemType || !quantity || !price) {
                alert('Please fill all fields');
                return;
            }
            
            const res = await fetch('/api/orders', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({
                    customerName,
                    phone,
                    items: [{type: itemType, quantity, price}]
                })
            });
            
            if (res.ok) {
                alert('Order created!');
                document.querySelectorAll('input').forEach(input => input.value = '');
                loadOrders();
                loadDashboard();
            }
        }
        
        // Update status
        async function updateStatus(orderId, status) {
            await fetch(`/api/orders/${orderId}/status`, {
                method: 'PATCH',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({status})
            });
            loadOrders();
            loadDashboard();
        }
        
        // Initial load
        loadDashboard();
        loadOrders();
    </script>
</body>
</html>
